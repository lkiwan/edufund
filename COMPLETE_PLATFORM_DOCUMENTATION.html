    Platform Name

    • 5. Database Schema

    Users Table

    `sql

    CREATE TABLE users (

    id INT PRIMARY KEY AUTO_INCREMENT,

    email VARCHAR(255) UNIQUE NOT NULL,

    password VARCHAR(255) NOT NULL,

    role ENUM('student', 'donor', 'admin', 'super-admin') DEFAULT 'student',

    full_name VARCHAR(255),

    phone VARCHAR(20),

    bio TEXT,

    profile_picture TEXT,

    cover_image TEXT,

    location VARCHAR(255),

    verified BOOLEAN DEFAULT 0,

    status ENUM('active', 'suspended', 'pending') DEFAULT 'active',

    profile_approved_at DATETIME,

    approved_by INT,

    rejection_reason TEXT,

    created_at DATETIME DEFAULT CURRENT_TIMESTAMP,

    updated_at DATETIME DEFAULT CURRENT_TIMESTAMP ON UPDATE CURRENT_TIMESTAMP

    );

    `

    Campaigns Table

    `sql

    CREATE TABLE campaigns (

    id INT PRIMARY KEY AUTO_INCREMENT,

    title VARCHAR(500) NOT NULL,

    description TEXT NOT NULL,

    goal_amount DECIMAL(10,2) NOT NULL,

    current_amount DECIMAL(10,2) DEFAULT 0,

    category VARCHAR(100),

    city VARCHAR(255),

    university VARCHAR(255),

    cover_image TEXT,

    status VARCHAR(50) DEFAULT 'draft',

    end_date DATETIME NOT NULL,

    created_at DATETIME DEFAULT CURRENT_TIMESTAMP,

    user_id INT NOT NULL,

    featured BOOLEAN DEFAULT 0,

    student_name VARCHAR(255),

    student_avatar TEXT,

    student_university VARCHAR(255),

    student_field VARCHAR(255),

    student_year VARCHAR(50),

    beneficiary_name VARCHAR(255),

    beneficiary_relationship VARCHAR(100),

    verification_status VARCHAR(50),

    trust_score INT DEFAULT 0,

    tags TEXT,

    allow_anonymous BOOLEAN DEFAULT 1,

    allow_comments BOOLEAN DEFAULT 1,

    withdrawn_amount DECIMAL(10,2) DEFAULT 0,

    approved_at DATETIME,

    approved_by INT,

    rejection_reason TEXT,

    moderation_notes TEXT,

    flagged BOOLEAN DEFAULT 0,

    flag_reason TEXT,

    FOREIGN KEY (user_id) REFERENCES users(id),

    FOREIGN KEY (approved_by) REFERENCES users(id)

    );

    `

    Donations Table

    `sql

    CREATE TABLE donations (

    id INT PRIMARY KEY AUTO_INCREMENT,

    campaign_id INT NOT NULL,

    user_id INT,

    donor_name VARCHAR(255),

    donor_email VARCHAR(255),

    donor_message TEXT,

    is_anonymous BOOLEAN DEFAULT 0,

    amount DECIMAL(10,2) NOT NULL,

    tip_amount DECIMAL(10,2) DEFAULT 0,

    currency VARCHAR(10) DEFAULT 'MAD',

    payment_method VARCHAR(50),

    status VARCHAR(50) DEFAULT 'completed',

    receipt_sent BOOLEAN DEFAULT 0,

    created_at DATETIME DEFAULT CURRENT_TIMESTAMP,

    flagged BOOLEAN DEFAULT 0,

    flag_reason TEXT,

    verified_by INT,

    verified_at DATETIME,

    FOREIGN KEY (campaign_id) REFERENCES campaigns(id) ON DELETE CASCADE,

    FOREIGN KEY (user_id) REFERENCES users(id) ON DELETE SET NULL

    );

    `

    Campaign Views Table

    `sql

    CREATE TABLE campaign_views (

    id INT PRIMARY KEY AUTO_INCREMENT,

    campaign_id INT NOT NULL,

    user_id INT NULL,

    ip_address VARCHAR(45),

    user_agent TEXT,

    referrer TEXT,

    viewed_at TIMESTAMP DEFAULT CURRENT_TIMESTAMP,

    session_id VARCHAR(255),

    FOREIGN KEY (campaign_id) REFERENCES campaigns(id) ON DELETE CASCADE,

    FOREIGN KEY (user_id) REFERENCES users(id) ON DELETE SET NULL,

    INDEX idx_campaign_id (campaign_id),

    INDEX idx_user_id (user_id),

    INDEX idx_viewed_at (viewed_at),

    INDEX idx_session_id (session_id)

    );

    `

    CREATE TABLE social_shares (

    `sql

    CREATE TABLE campaign_followers (

    id INT PRIMARY KEY AUTO_INCREMENT,

    campaign_id INT NOT NULL,

    created_at TIMESTAMP DEFAULT CURRENT_TIMESTAMP,

    FOREIGN KEY (user_id) REFERENCES users(id) ON DELETE CASCADE,

    • 14. Troubleshooting

    Common Issues

    #### Database Connection Failed

    Error:     Error: connect ECONNREFUSED Solution:
  • Ensure MySQL is running
  • Check database credentials in .env
  • Verify database exists: node create-database.js
  • Check MySQL port (default 3306)

    • #### Frontend Cannot Connect to Backend

    Error:     Failed to fetch or Network Error Solution:
  • Ensure backend is running on port 3001
  • Check Vite proxy in vite.config.mjs
  • Verify CORS settings in server.js
  • Clear browser cache

    • #### Image Upload Fails

    Error:     Upload failed or File too large Solution:
  • Check uploads/campaigns/ directory exists
  • Verify write permissions
  • Ensure file size < 5MB
  • Check file type (JPG/PNG only)

    • #### Trending Campaigns Not Showing

    Issue: Empty trending section Cause: No recent activity (views, donations, shares) Solution:
  • Add some test data
  • Track views manually
  • Make test donations
  • Share campaigns
  • Adjust time period (try "week" or "month")

    • #### Analytics Not Loading

    Error:     Failed to fetch analytics Solution:
  • Check if analytics tables exist
  • Run create-analytics-tables.sql
  • Verify campaign has some activity
  • Check console for SQL errors
